class Main {
  public static void main(String args[]) {
    int media1;
    int media2;
    int mediamaismedia;

    media1 = 8 + 9 + 7 / 3;
    media2 = 4 + 5 + 6 / 3;
    mediamaismedia = media1 + media2 / 2;
    System.out.println("A média dos números 8, 9 e 7 é: "+ media1);
    System.out.println("A média dos números 4, 5 e 6 é: "+ media2);
    System.out.println("A média da media 1 e da média 2 é: "+ mediamaismedia);
  }
}
